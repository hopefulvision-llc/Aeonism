```python
def calculate_measurement_day_positions(year_start_day=0):
```

### Conversion Algorithm (Gregorian to Aeonist)
```python
def gregorian_to_aeonist(day_of_year, measurement_days):
    """
    Convert Gregorian day of year to Aeonist calendar
    day_of_year: 1-365
    measurement_days: list of 5 measurement day positions
    """
    # Remove measurement days from calculation
    observable_day = day_of_year
    for m_day in sorted(measurement_days):
        if day_of_year > m_day:
            observable_day -= 1
        elif day_of_year == m_day:
            return {
                "type": "measurement_day",
                "measurement_number": measurement_days.index(m_day) + 1,
                "month": 0,
                "day": 0,
                "agent": "MeasurementDay"

    # Check for Spiral Day (day 365)
    if observable_day > 364:
        return {
            "type": "spiral_day",
            "month": 14,
            "day": 1,
            "agent": "SpiralDay"

    # Calculate month and day (1-indexed)

        "VeilKeeper"

    return {
        "type": "regular_day",

        "day": aeonist_day,

    }
```
